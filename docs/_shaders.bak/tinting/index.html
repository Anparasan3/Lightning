Tinting

You can define a color for your texture by simply setting it's color.

this.tag('Element').color = 0xffff00ff

Live Demo

class TintingDemo extends lng.Application {
    static _template() {
        return {
            x: 50,
            y: 50,
            Original: {
                src: 'http://i.imgur.com/bgDyE.jpg',
                pivot: 0, //sets the point of transformations to 0,0 (default is 0.5, 0.5)
                scale: 0.1, //scales the photo down to 10%
            },
            Tinted: {
                x: 205,
                src: 'http://i.imgur.com/bgDyE.jpg',
                pivot: 0,
                scale: 0.1,
                color: 0xffff00ff, //tints the image with magenta
            },
            Tinted2: {
                y: 137,
                src: 'http://i.imgur.com/bgDyE.jpg',
                pivot: 0,
                scale: 0.1,
                color: 0xff00ff00, //tints the image with red
            },
            Tinted3: {
                x: 205,
                y: 137,
                src: 'http://i.imgur.com/bgDyE.jpg',
                pivot: 0,
                scale: 0.1,
                color: 0xff0000ff, //tints the image with blue
            },
        }
    }
}

const options = {
    stage: {
        w: 600,
        h: 600,
        clearColor: 0x00000000,
        canvas2d: false,
        useImageWorker: false,
    },
    debug: true,
}

const app = new TintingDemo(options)
document.body.appendChild(app.stage.getCanvas())
