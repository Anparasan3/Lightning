Fires an event downwards

class MyBroadcastingClass extends lng.Application {
    static _template() {
        return {
            Example:{ type: MyReceivingTextClass, x: 100, y: 100 }
        };
    }

    _handleEnter(){
        this.broadcast('myBroadCast', { myNewText: 'BroadCast Received!' });
    }
}

class MyReceivingTextClass extends lng.Component {
    static _template() {
        return {
            broadcasts:{
                myBroadCast: '_handleMyBroadCast'
            },
            Label:{  text:{ text: 'Hello World!'}}
        };
    }
    
    _handleMyBroadCast(myNewText){
        console.log('received broadcast', myNewText);
        this.tag('Label').patch({ text:{ text: myNewText } });
    }
}
