Player

Lightning SDK has a player component that can be used to stream media.

This component is an abstraction of the Mediaplayer.

You can use it in your projects, or extend and customize it according to your app requirements, or even build your own based on it.

export default class MyPlayer extends ux.tools.player.Player {

    static _template() {
        let newTemplate = super._template()
        //your template changes here

        return newTemplate
    }
}

You can add events and customize them. Check Mediaplayer's _init() for a full list.

These events will be fired in your consumer (in this case, your Player) with the prefix $mediaplayer + Event (camelCased), so you can add methods to listen to them.

$mediaplayerPause() {
    //do something when the player pauses
}

class AppDefinition extends ux.AppDefinition {
    static get identifier() {
        return "com.metrological.app.Example";
    }
    getAppViewType() {
        return PlayerDemo;
    }
}


class PlayerDemo extends lng.Application {
    static _template() {
        return {
            x: 50, y: 50,
            Player: {
                type: ExamplePlayer
            },
            Button: {
                x: 450,
                type: ExampleButton,
                buttonText: 'Maximize'
            }
        }
    }

    _init() {
        this._setState('InlinePlayer')
    }

    static _states() {
        return [
             class InlinePlayer extends this {
                $enter() {

                }
                _getFocused() {
                    return this.tag('Button')
                }
            },
            class PlayerFullScreen extends this {
                $enter() {
                    this.tag('Player').fire('setFullscreen', true)
                }
                $exit() {
                    this.tag('Player').fire('setFullscreen', false)
                }
                _getFocused() {
                    return this.tag('Player')
                }
            }
        ]
    }

    _setFocusSettings(settings) {
        settings.mediaplayer.consumer = this.tag('Player')
        settings.mediaplayer.videoPos = [50, 50, 300, 300]
    }
    
    static getFonts() {
        return []
    }
}

class ExampleButton extends lng.Component {
    static _template() {
        return {
            color: 0xff1f1f1f,
            texture: lng.Tools.getRoundRect(150, 40, 4),
            Label: {
                x: 75, y: 22, mount: .5, color: 0xffffffff, text: { fontSize: 20 }
            }
        }
    }
    _init() {
        this.tag('Label').patch({ text: { text: this.buttonText }})
    }
    _focus() {
        this.color = 0xffffffff
        this.tag('Label').color = 0xff1f1f1f
    }
    _unfocus() {
        this.color = 0xff1f1f1f
        this.tag('Label').color = 0xffffffff
    }
}

class ExamplePlayer extends ux.tools.player.Player {

    setFullscreen(fullscreen) {
        this.playerFullscreen = fullscreen
    }

    getMediaplayerSettings() {
        return {
            videoPos: this.playerFullscreen ? [0, 0, 600, 600] : [50, 50, 300, 300],
            stream: { src: 'http://video.metrological.com/boat.mp4' },
        }
    }
}

const options = {
    stage: {
        w: 600,
        h: 600,
        clearColor: 0x00000000,
        canvas2d: false,
        useImageWorker: false,
    },
    debug: true,
    keys: {
        13: "Enter",
        38: "Up",
        40: "Down",
        37: "Left",
        39: "Right",
        9: "Back",
        8: "Back",
        93: "Back",
        174: "Back",
    },
}

const bootstrap = new ux.Ui(options);
bootstrap.startApp(new AppDefinition(options));
const canvas = bootstrap.stage.getCanvas();
document.body.appendChild(canvas);

window.app = bootstrap;
